41int Field::update()
42{
43 int k = heightBlocks - 1, numberOfLines = 0;
44 for (int i = heightBlocks - 1; i >= 0; i--)
45 {
46 int count = 0;
47 for (int j = 0; j < widthBlocks; j++)
48 {
49 if (field[i][j])
50 count++;
51 field[k][j] = field[i][j];

53 if (count < widthBlocks)
54 {
55 k--;

60 }

64 {
65 ups[i] -= numberOfLines;
