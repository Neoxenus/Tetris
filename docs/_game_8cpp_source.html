63void Game::eventListener()
64{
65 sf::Event event;
66 while (window.pollEvent(event))
67 {
68 if (event.type == sf::Event::Closed)
69 window.close();
70 if (event.type == sf::Event::LostFocus)
71 pause = true;
72 if (event.type == sf::Event::MouseButtonReleased && event.mouseButton.button == sf::Mouse::Left)
73 {
74 this->UIMenu();
75 }
76 if (event.type == sf::Event::KeyPressed)
77 {
78 if (event.key.code == sf::Keyboard::Escape && isGameActive)
79 {
80 pause = !pause;
81 }
82 if (isBot || pause)
83 continue;
84 if (isGameActive)
85 {
86 if (event.key.code == sf::Keyboard::Up)
87 rotate = true;
88 else if (event.key.code == sf::Keyboard::Left)
89 dx = -1;
90 else if (event.key.code == sf::Keyboard::Right)
91 dx = 1;
92 }
93 }

95 }

111void Game::mainUpdate()
112{
113 time = clock.getElapsedTime().asSeconds();
114 clock.restart();
115 timer += time; mainTimer += time;
116 statistics.SetStatistics(mainTimer, static_cast<int>(constants::Stat::TIME));
117 mainField.updateDynamicDelay(mainTimer);
118 if (isBot && bot.readyToFall())
119 mainField.SetDelay(mainField.GetDelay() / std::pow(50, statistics.GetStatistics(static_cast<int>(constants::Stat::DIFFICULTY))));
120 if (!isBot && sf::Keyboard::isKeyPressed(sf::Keyboard::Down))
121 mainField.SetDelay(mainField.GetDelay() / 10);
122 //left and right moving

124

126 if (rotate)
127 currentTetromino.rotate();
128 //fps counter
129
130 //tick of timer

132 this->tickOfTimer();
